Module Wodoc.Config

type entry = {
  1. label : string;
    (*

    visible text

    *)
  2. path : string;
    (*

    href relative to the version root (filled with base)

    *)
  3. current : string;
    (*

    data-wodoc-page id (highlighted on its own pages)

    *)
}
type section = {
  1. heading : string;
    (*

    the <h3> label of this left-nav block

    *)
  2. api : bool;
    (*

    api-section -> class "api-nav"; else "api-nav manual-nav"

    *)
  3. entries : entry list;
}
type cs_side = {
  1. side : string;
    (*

    key, e.g. "server" — body class wodoc-server, switch id

    *)
  2. lib : string;
    (*

    odoc subtree dir / Nav.api ~lib, e.g. "eliom.server"

    *)
  3. indexdoc : string;
    (*

    curated index path (relative to the build cwd)

    *)
  4. heading : string;
    (*

    API-nav heading + switch-button label, e.g. "Server API"

    *)
  5. wrapper : string;
    (*

    Nav.api ~wrapper / page wrapper module, e.g. "Eliom"

    *)
  6. skip : string list;
    (*

    section titles to omit (Nav.api ~skip)

    *)
}

one API "side" of a client/server project (eliom/toolkit/start): the <pkg>.server / <pkg>.client / <pkg>.ppx library, whose curated index becomes a section-grouped module nav (via Nav.api). The side also drives a body colour class (wodoc-<side>) and a switch button.

type t = {
  1. project : string;
    (*

    package / project id, e.g. "ocsipersist"

    *)
  2. title : string;
    (*

    sub-project label by the logo, e.g. "Ocsipersist"

    *)
  3. pub : string;
    (*

    absolute publish base for the version selector, e.g. "/ocsipersist"

    *)
  4. menu_current : string;
    (*

    project id highlighted in the shared top menu

    *)
  5. packages : string list;
    (*

    odoc output subtrees to assemble, in order

    *)
  6. landing : string;
    (*

    index.html redirect target, e.g. "ocsipersist/index.html"

    *)
  7. highlight : string option;
    (*

    project highlight.js to ship, if any

    *)
  8. profile : string option;
    (*

    dune build profile (e.g. "release")

    *)
  9. odoc_driver : string option;
    (*

    when set, build the API with odoc_driver <pkg> --remap (the engine ocaml.org uses) instead of dune build @doc: needed for a client/server package whose <pkg>.server/<pkg>.client libraries share module names and would collide under dune build @doc (eliom, ocsigen-toolkit, ocsigen-start).

    *)
  10. doc_manual : bool;
    (*

    also build the @doc-manual alias (examples)

    *)
  11. manual_files : string option;
    (*

    package dir to receive manual/files (examples, images)

    *)
  12. siblings : (string * string list) list;
    (*

    resolve-refs sibling table

    *)
  13. nav : section list;
  14. client_server : cs_side list;
    (*

    when non-empty, the project is client/server: instead of one nav from nav, wodoc builds a per-side API nav from each side's curated index, shows a client/server switch, and colours the body by side. The manual nav is shared (from manual_menu, else from nav's manual sections).

    *)
  15. manual_menu : string option;
    (*

    a wiki menu.wiki file driving the shared manual nav (Nav.manual); used by client/server projects whose manual nav comes from wikidoc

    *)
  16. hosted : (string * (string * bool * string)) list;
    (*

    cross-project resolve-refs table (resolve-refs --hosted): package -> (deploy dir, multi-library?, wrapper module). Rewrites sibling Ocsigen projects' ocaml.org xrefs to relative links into their wodoc docs.

    *)
  17. mld_dir : string option;
    (*

    direct-mld build (a manual-only / archived project with no dune build @doc): compile every .mld in this dir straight with odoc (preprocess -> compile -> link -> html-generate). The pages are the manual; the landing index.html is a real page (no redirect).

    *)
  18. mld_package : string;
    (*

    odoc --package for the direct-mld compile (the src subtree)

    *)
  19. flat : bool;
    (*

    assemble --flat (content straddling odoc's preamble boundary)

    *)
  20. anchor_menu : string option;
    (*

    left nav from a single page's in-page anchors (Nav.anchors), e.g. a one-page manual whose menu links to #section anchors

    *)
  21. static_copy : (string * string) list;
    (*

    verbatim copies into the output: (source path, dest under <out>) — e.g. a frozen API snapshot, or a manual image

    *)
}
val parse_entry : Sexp.t -> entry
val parse_section : bool -> Sexp.t -> section
val parse_nav : Sexp.t list -> section list
val parse_siblings : Sexp.t list -> (string * string list) list
val parse_client_server : Sexp.t list -> cs_side list
val parse_hosted : Sexp.t list -> (string * (string * bool * string)) list
val parse_static_copy : Sexp.t list -> (string * string) list
val of_string : string -> t
